function formatTime(ms) {

  let s = Math.floor(ms / 1000);

  let m = Math.floor(s / 60);

  s = s % 60;

  let h = Math.floor(m / 60);

  m = m % 60;

  return `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;

}

        <input type="text" id="userInput" placeholder="${placeholder}" onkeydown="if(event.key==='Enter')submitAnswer()" autocomplete="off" autocapitalize="off" />

        <button class="send-btn" onclick="submitAnswer()">➤</button>

      </div>

    `;

  }

  

  html += `<div class="scroll-anchor" id="scrollAnchor"></div>`;

  

  app.innerHTML = html;

  

  // Auto-scroll

  setTimeout(() => {

    const anchor = document.getElementById('scrollAnchor');

    if (anchor) anchor.scrollIntoView({ behavior: 'smooth' });

    const input = document.getElementById('userInput');

    if (input) input.focus();

  }, 50);

}


function escHtml(text) {

  return text.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/\n/g,'<br>');

}


function startStop(idx) {

  state.currentStop = idx;

  state.hintsUsed = 0;

  state.puzzleSolved = false;

  state.waitingForReady = false;

  state.waitingForAnswer = false;

  

  const stop = STOPS[idx];

  if (!stop) return;

  

  // Queue messages with delays

  state.msgQueue = [...stop.messages];

  showNextQueued();

}


function showNextQueued() {

  if (state.msgQueue.length === 0) {

    // All messages shown, now wait for input

    const stop = STOPS[state.currentStop];

    if (stop.type === "intro") {

      state.waitingForReady = false;

      state.waitingForAnswer = true;

    } else {

      state.waitingForAnswer = true;

    }

    render();

    return;

  }

  

  const msg = state.msgQueue.shift();

  state.displayedMsgs.push(msg);

  render();

  

  if (state.msgQueue.length > 0) {

    setTimeout(showNextQueued, 600);

  } else {

    showNextQueued();

  }

}


function submitAnswer() {

  const input = document.getElementById('userInput');

  if (!input) return;

  const val = input.value.trim();

  if (!val) return;

  

  const stop = STOPS[state.currentStop];

  

  // Intro stop

  if (stop.type === "intro") {

    if (stop.accept.includes(val)) {

      state.displayedMsgs.push({ from: 'player', text: stop.playerResponse });

      state.waitingForAnswer = false;

      // Start timer

      state.timerRunning = true;

      state.startTime = Date.now();

      state.totalPaused = 0;

      // Move to stop 1 after brief delay

      render();

      setTimeout(() => {

        state.displayedMsgs = [];

        startStop(1);

      }, 800);

    } else {

      state.displayedMsgs.push({ from: 'player', text: val });

      state.displayedMsgs.push({ from: 'narrator', text: "Head to the Cy Young Statue outside Cabot Center and type START when you're ready!" });

      render();

    }

    return;

  }

  

  // Check for READY responses between stops

  if (state.waitingForReady) {

    const readyWords = ["READY", "ready", "Ready", "DONE", "done", "Done"];

    if (readyWords.includes(val)) {

      state.displayedMsgs.push({ from: 'player', text: val });

      state.waitingForReady = false;

      

      // If it's the lunch stop DONE, unpause timer

      if (stop.id === 4 && state.timerPaused) {

        state.totalPaused += Date.now() - state.pauseStart;

        state.timerPaused = false;

        state.pauseStart = null;

      }

      

      render();

      setTimeout(() => {

        state.displayedMsgs = [];

        startStop(state.currentStop + 1);

      }, 600);

    } else {

      state.displayedMsgs.push({ from: 'player', text: val });

      state.displayedMsgs.push({ from: 'narrator', text: "Type READY when you've arrived at the next location! (Or DONE if you're at the lunch break.)" });

      render();

    }

    return;

  }

  

  // Check answer

  if (state.waitingForAnswer) {

    state.displayedMsgs.push({ from: 'player', text: val });

    

    if (stop.answer.includes(val) || stop.answer.map(a => a.toUpperCase()).includes(val.toUpperCase())) {

      state.puzzleSolved = true;

      state.waitingForAnswer = false;

      

      // Last stop?

      if (state.currentStop === STOPS.length - 1) {

        // Finale!

        state.elapsed = Date.now() - state.startTime - state.totalPaused;

        state.timerRunning = false;

        state.finished = true;

        for (const msg of FINALE) {

          state.displayedMsgs.push(msg);

        }

        state.displayedMsgs.push({ from: 'system', text: `⏱ Your time: ${formatTime(state.elapsed)}` });

        render();

        return;

      }

      

      // Success message

      state.displayedMsgs.push({ from: 'system', text: "✅ Correct!" });

      state.displayedMsgs.push({ from: 'narrator', text: stop.successMsg });

      state.waitingForReady = true;

      

      // Auto-pause for lunch stop

      if (STOPS[state.currentStop + 1] && STOPS[state.currentStop + 1].id === 4) {

        // The next stop is the lunch break — it will auto-pause when it loads

      }

      if (stop.id === 3) {

        // After stop 3 success, the next stop (4) is lunch — pause timer when they get there

      }

      

      render();

    } else {

      state.displayedMsgs.push({ from: 'wrong', text: "Hmm, that's not quite right. Try again! Use a hint if you're stuck — no penalties, this is just for fun." });

      render();

    }

    return;

  }

}


function useHint(idx) {

  const stop = STOPS[state.currentStop];

  if (!stop || !stop.hints || idx >= stop.hints.length) return;

  state.hintsUsed = idx + 1;

  state.displayedMsgs.push({ from: 'hint', text: stop.hints[idx] });

  render();

}


function showAnswer() {

  const stop = STOPS[state.currentStop];

  if (!stop) return;

  const ans = stop.answer[0];

  state.displayedMsgs.push({ from: 'hint', text: `The answer is: ${ans}\n\nType it in to continue!` });

  state.hintsUsed = stop.hints ? stop.hints.length : 0;

  render();

}


function togglePause() {

  if (state.timerPaused) {

    state.totalPaused += Date.now() - state.pauseStart;

    state.timerPaused = false;

    state.pauseStart = null;

  } else {

    state.timerPaused = true;

    state.pauseStart = Date.now();

  }

  render();

}


// Timer update

setInterval(() => {

  if (state.timerRunning && !state.timerPaused) {

    const el = document.getElementById('timerDisplay');

    if (el) {

      const elapsed = Date.now() - state.startTime - state.totalPaused;

      el.textContent = formatTime(elapsed);

    }

  }

}, 1000);


// Auto-pause on lunch stop

const origStartStop = startStop;

startStop = function(idx) {

  origStartStop(idx);

  if (STOPS[idx] && STOPS[idx].id === 4 && !state.timerPaused) {

    state.timerPaused = true;

    state.pauseStart = Date.now();

    render();

  }

};


// Init

startStop(0);
